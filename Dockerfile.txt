# Build stage using Maven and JDK 17
FROM maven:3.8.5-openjdk-17 AS build
WORKDIR /app

# Copy the pom.xml and the source code
COPY pom.xml .
COPY src ./src

# Build the application and skip tests to save time on Render
RUN mvn clean package -DskipTests

# Runtime stage using a slim JRE 17 image
FROM eclipse-temurin:17-jre-alpine
WORKDIR /app

# Copy the JAR from the build stage
# Based on your pom.xml, the jar is cafe-0.0.1-SNAPSHOT.jar
COPY --from=build /app/target/cafe-0.0.1-SNAPSHOT.jar app.jar

# Expose the port Spring Boot runs on
EXPOSE 8080

# Run the application
ENTRYPOINT ["java", "-jar", "app.jar"]
